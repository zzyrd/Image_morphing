def rect_to_bb(rect):
    # take a bounding predicted by dlib and convert it 
    # to the format (x, y, w, h) as we would normally do
    x = rect.left()
    y = rect.top()
    w = rect.right() - x
    h = rect.bottom() - y
    
    return (x, y, w, h)

def shape_to_np(shape, dtype="int"):
    # initialize the list of (x, y)-coordinates
    coords = np.zeros((68, 2), dtype=dtype)
    # loop over the 68 facial landmarks and convert them
    # to a 2-tuple of (x, y)-coordinates
    for i in range(0, 68):
        coords[i] = (shape.part(i).x, shape.part(i).y)
    # return the list of (x, y)-coordinates
    return coords

# for each face in rects: rects contains all the bounding boxes for all faces.
for (i, rect) in enumerate(rects):
    # find the facial landmarks, then convert it to a numpy array
    points = predictor(gray, rect)
    points = shape_to_np(points)  

    # convert dlib's rectangle to a opencv-style bounding box (x,y,w,h)
    (x, y, w, h) = rect_to_bb(rect)
    cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 3)

    # show the face number
    cv2.putText(image, "Face #{}".format(i + 1), (x - 10, y - 10),
                cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
    
    # loop over the x-y coordinates for the facial landmarks
    for (x,y) in points:
        cv2.circle(image, (x,y), 3, (0,0,255), -1)

# Draw delaunay triangles
def draw_delaunay(img, subdiv, delaunay_color):

    triangleList = subdiv.getTriangleList();
#     size = img.shape
#     r = (0, 0, size[1], size[0])
    
    for t in triangleList :
        pt1 = (t[0], t[1])
        pt2 = (t[2], t[3])
        pt3 = (t[4], t[5])
        cv2.line(img, pt1, pt2, delaunay_color, 2, cv2.LINE_AA)
        cv2.line(img, pt2, pt3, delaunay_color, 2, cv2.LINE_AA)
        cv2.line(img, pt3, pt1, delaunay_color, 2, cv2.LINE_AA)


# # Draw voronoi diagram
def draw_voronoi(img, subdiv) :
    (facets, centers) = subdiv.getVoronoiFacetList([])
    for i in range(0,len(facets)):
        ifacet_arr = []
        for f in facets[i] :
            ifacet_arr.append(f)
        ifacet = np.array(ifacet_arr, np.int)
        color = (random.randint(0, 255), random.randint(0, 255), random.randint(0, 255))

        cv2.fillConvexPoly(img, ifacet, color, cv2.LINE_AA);
        ifacets = np.array([ifacet])
        cv2.polylines(img, ifacets, True, (0, 0, 0), 1, cv2.LINE_AA, 0)
        cv2.circle(img, (centers[i][0], centers[i][1]), 2, (255, 255, 255), 2, cv2.LINE_AA)
